\newcommand{\Current}[3]{
	\newRow
	\node[label] at (\x,\y) {$#1$} ;
	\draw[connect,color=red] (\x+.5,\y)	 -- (\x+1.5,\y) ;
	\node[label] at (\x+3.,\y) {\texttt{#2}} ;
	\node[label] at (\x+4.5,\y) {\texttt{#3}} ;
}

	\node[label] at (\x+5.5,\y) {\checked{#1}	} ;
	\node[label] at (\x+5.9,\y) {\pppp{#2}		} ;	
}
